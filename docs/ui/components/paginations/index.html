在使用 Backgrid Paginator 时不要生成前往首页和末页的触发器。

// JavaScript

var paginator = new Backgrid.Extension.Paginator({
    controls: {
      rewind: null,     // 不生成首页
      fastForward: null // 不生成末页
    }
  });
